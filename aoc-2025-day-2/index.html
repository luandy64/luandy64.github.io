    So my first thought at an approach here is to use a regex and a lookahead pattern. I want to turn each number into either itself because it has a matching lookahead or a nil. So like 11 should “pass” the lookahead because the “needle” of 1 does appear again. But for 12, this is not true.

    Maybe this is called a negative lookahead or maybe the behavior I’m describing doesn’t exist at all and I need to do it myself.

    So a quick search tells me that this is actually a positive lookahead and that I probably need to dynamically generate these patterns and then run it against my input.

    So now the thought has hit me that trying to construct these things is a little too much work. Can I just take a number, make it a string, split it in half and check if the two halves are equal?

    A cleaned up solution

    (defun input->ranges (input-str)
  (mapcar (lambda (token)
            (mapcar 'string-to-number
                     (split-string token "\-")))
          (split-string input-str
                        "\,")))

(defun make-range (range)
  (apply 'number-sequence range))

(defun part-1 (a)
  (let* ((str-a        (number-to-string a))
         (middle-index (/ (length str-a) 2))
         (invalid-str  (concat (substring str-a 0 middle-index)
                               (substring str-a 0 middle-index))))
    (string-to-number invalid-str)))

(defun do-part-1 (curr-range)
  (mapcar #'part-1
            curr-range))

(defun do-process (do-fn range)
  (let ((current-range (make-range range)))
    (flatten-tree
     (cl-remove-duplicates
      (remove nil
              (funcall do-fn current-range))
      :test 'equal))))

(defun loop-n-reduce (input process-range-fn)
  (let* ((ranges (input->ranges input)))
    (seq-reduce
     #'+
     (flatten-tree
      (cl-loop
       for range in ranges
       append (cl-destructuring-bind (start end) range
                (let* ((candidates (do-process process-range-fn
                                               range)))
                  (seq-filter (lambda (candidate)
                                (<= start candidate end))
                              candidates)))))
     0)))

(defun make-stems (a)
  (let* ((str-a        (number-to-string a))
         (middle-index (/ (length str-a) 2)))
    (mapcar (lambda (stop-index)
              (string-to-number
               (substring str-a 0 stop-index)))
            (number-sequence 1 middle-index))))

(defun stem->patterns (stem)
  (let* ((str-stem     (number-to-string stem))
         (middle-index (/ (length str-stem) 2))
         (curr-num str-stem)
         (return-val '()))
    (while (<= (string-to-number curr-num) end)
      (if (<= start (string-to-number curr-num) end)
          (push (string-to-number curr-num) return-val))
      (setq curr-num (concat curr-num str-stem)))
    return-val))

(defun do-part-2 (curr-range)
  (let ((stems (cl-remove-duplicates
                (flatten-list
                 (mapcar #'make-stems
                           curr-range)))))
    (mapcar #'stem->patterns
              stems)))

(loop-n-reduce example-input 'do-part-1)
(loop-n-reduce example-input 'do-part-2)

    Cleaned up with dash.el

    I’m used to the grand, unified design of Clojure and the elisp/common-lisp looks kinda crazy.

    Thankfully there is the dash.el library that brings a ton of Clojure style functions to elisp and most importantly the threading macros.

    (defun input->ranges (input-str)
  (-map (lambda (token)
          (-map #'string-to-number
                  (split-string token
                                "\-")))
        (split-string input-str
                      "\,")))

(defun make-range (range)
  (apply #'number-sequence range))

(defun part-1 (a)
  (let* ((str-a  (number-to-string a))
         (substr (->> (-> str-a
                          length
                          (/ 2))
                      (substring str-a
                                 0))))
    (-> substr
        (concat substr)
        string-to-number)))

(defun do-part-1 (curr-range)
  (-map #'part-1
          curr-range))

(defun do-process (do-fn range)
  (->> range
       make-range
       (funcall do-fn)
       -flatten
       -distinct))

(defun loop-n-reduce (input process-range-fn)
  (->> (cl-destructuring-bind (start end) range
         (->> range
              (do-process process-range-fn)
              (-filter (lambda (num)
                         (<= start num end)))))
       (cl-loop
        for range in (input->ranges input)
        append)
       -flatten
       (-reduce-from #'+ 0)))

(defun make-stems (a)
  (let ((str-a (number-to-string a)))
    (->> (-> str-a
             length
             (/ 2))
         (number-sequence 1)
         (-map (lambda (stop-index)
                 (->> stop-index
                      (substring str-a
                                 0)
                      string-to-number))))))

(defun stem->patterns (stem)
  (let* ((str-stem     (number-to-string stem))
         (middle-index (-> str-stem
                           length
                           (/ 2)))
         (curr-num     str-stem)
         return-val)
    (while (<= (string-to-number curr-num) end)
      (-> curr-num
          string-to-number
          (push return-val))
      (setq curr-num
            (concat curr-num
                    str-stem)))
    return-val))

(defun do-part-2 (curr-range)
  (->> curr-range
       (-map #'make-stems)
       -flatten
       -distinct
       (-map #'stem->patterns)))

(loop-n-reduce example-input 'do-part-1)
(loop-n-reduce example-input 'do-part-2)
